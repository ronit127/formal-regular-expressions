Interactive PyScript Example

Enter a number and click the button:

from pyscript import Element # Define a function that will be triggered by the button click def calculate_square(event): num = int(Element("num-input").element.value) # Get value from input field result = num ** 2 Element("result").write(f"The square of {num} is {result}") # Use the 'add_event_listener' method to bind the function to the button click event Element("calc-btn").add_event_listener("click", calculate_square)
